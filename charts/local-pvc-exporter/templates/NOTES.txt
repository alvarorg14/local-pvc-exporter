  kubectl -n {{ .Release.Namespace }} get pods -l "app.kubernetes.io/name={{ include "local-pvc-exporter.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

  kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "local-pvc-exporter.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
  curl -s localhost:{{ .Values.service.port }}/metrics | head
